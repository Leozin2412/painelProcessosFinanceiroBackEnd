create table public.timesheets (
  id bigint generated by default as identity not null,
  ident_sinistro bigint not null,
  codigo_sinistro text null,
  regulador_prestador text null,
  tp_incidencia text null,
  dt_inicial timestamp with time zone null,
  total_decimal numeric(12, 4) null,
  descricao_tarefa text null,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  ident_timesheet bigint null,
  dados_gclaims jsonb null default '{}'::jsonb,
  dt_final timestamp with time zone null,
  total_minutos integer null,
  hash_unico text null,
  ident_compania bigint null,
  nro_correlativo integer null,
  ident_seguradora bigint null,
  seguradora text null,
  dat_criacao_sinistro timestamp with time zone null,
  data_ocorrencia timestamp with time zone null,
  ident_operacao bigint null,
  operacao text null,
  ident_cobertura bigint null,
  cobertura text null,
  ident_incidencia bigint null,
  ident_tipo_incidencia bigint null,
  incidencia text null,
  possui_cobranca boolean null,
  cobrado boolean null,
  dt_lancamento timestamp with time zone null,
  unidade_cobranca text null,
  valor_por_hora numeric(20, 4) null,
  valor_cobrado numeric(20, 4) null,
  tipo_pessoa text null,
  ident_regulador_prestador bigint null,
  honorario_gerado boolean null,
  honorario_faturado boolean null,
  sinistros_fechados boolean null,
  sinistros_cancelados boolean null,
  tipo_timesheet text null,
  subtipo_timesheet text null,
  descricao_timesheet text null,
  data_criacao_hon timestamp with time zone null,
  quantidade numeric(20, 4) null,
  valor_unitario numeric(20, 4) null,
  valor_ajuste numeric(20, 4) null,
  data_despesa timestamp with time zone null,
  nro_documento text null,
  nro_erp text null,
  nro_nota text null,
  taxa numeric(20, 4) null,
  valor_bruto numeric(20, 4) null,
  valor_liquido numeric(20, 4) null,
  valor_nota numeric(20, 4) null,
  foi_cobrado boolean null,
  data_cobranca timestamp with time zone null,
  foi_pago boolean null,
  data_pagamento timestamp with time zone null,
  descricao_fatura text null,
  valor_ajuste_fatura numeric(20, 4) null,
  data_criacao timestamp with time zone null,
  numero_aviso text null,
  data_modificacao_timesheet timestamp with time zone null,
  constraint timesheets_pkey primary key (id),
  constraint timesheets_hash_unico_key unique (hash_unico),
  constraint timesheets_ident_timesheet_key unique (ident_timesheet)
) TABLESPACE pg_default;

create unique INDEX IF not exists idx_timesheets_ident_timesheet_unique on public.timesheets using btree (ident_timesheet) TABLESPACE pg_default
where
  (ident_timesheet is not null);

create index IF not exists idx_timesheets_ident_sinistro on public.timesheets using btree (ident_sinistro) TABLESPACE pg_default;

create index IF not exists idx_timesheets_dt_inicial on public.timesheets using btree (dt_inicial) TABLESPACE pg_default;

create index IF not exists idx_timesheets_codigo_sinistro on public.timesheets using btree (codigo_sinistro) TABLESPACE pg_default;

create unique INDEX IF not exists idx_timesheets_hash_unico_unique on public.timesheets using btree (hash_unico) TABLESPACE pg_default
where
  (hash_unico is not null);

create index IF not exists idx_timesheets_codigo_sinistro_cover on public.timesheets using btree (codigo_sinistro) INCLUDE (
  total_minutos,
  total_decimal,
  tipo_timesheet,
  sinistros_cancelados
) TABLESPACE pg_default;

create trigger trg_timesheet_faturado
after INSERT
or
update on timesheets for EACH row
execute FUNCTION resetar_status_bh_por_timesheet ();
